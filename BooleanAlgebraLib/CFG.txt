Prog ::= Or
Or ::= And or Or
	| And
And ::= Not and And
	| Not
Not ::= not Not
	| Expression
Expression ::= leftParenthesis Or rightParenthesis
	| identifier
	| true
	| false

And ::= Not (and And)*

	| Expression

	| identifier
	| true
	| false
